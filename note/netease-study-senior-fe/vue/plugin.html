# Mixin 模式

  • Mixin
    • 通过原型继承，扩展对象，实现复用
  • Vue.mixin
    • 全局注册混入
    • 会影响所有创建的 Vue 实例
  • Options.mixin
    • 局部混入
  • 同名钩子函数合并为数组，混入对象钩子在组件自身钩子之前调用
  • components、directives、methods，合并为一个对象，组件对象覆盖混入对象

# 插件

  • Vue.use

    • 安装 Vue.js 插件

    • 接收一个函数或提供 install方法的对象作为参数












       

       







      // https://github.com/vuejs/vue/blob/dev/src/core/global-api/use.js
export function initUse (Vue: GlobalAPI) {
  Vue.use = function (plugin: Function | Object) {
    const installedPlugins = (this._installedPlugins || (this._installedPlugins = []))
    if (installedPlugins.indexOf(plugin) > -1) {
      return this
    }

    // additional parameters
    const args = toArray(arguments, 1)
    args.unshift(this)
    if (typeof plugin.install === 'function') {
      plugin.install.apply(plugin, args)
    } else if (typeof plugin === 'function') {
      plugin.apply(null, args)
    }
    installedPlugins.push(plugin)
    return this
  }
}

      // https://github.com/vuejs/vuex/blob/dev/src/mixin.js
export default function (Vue) {
  const version = Number(Vue.version.split('.')[0])

  if (version >= 2) {
    Vue.mixin({ beforeCreate: vuexInit })
  } else {
    /* ... */
  }

  function vuexInit () {
    const options = this.$options
    // store injection
    if (options.store) {
      this.$store = typeof options.store === 'function'
        ? options.store()
        : options.store
    } else if (options.parent && options.parent.$store) {
      this.$store = options.parent.$store
    }
  }
}
